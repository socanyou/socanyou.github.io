import java.util.Map;

import com.opensymphony.xwork2.ActionSupport;

import domain.Goods;
public class GoodsAction2 extends ActionSupport {

    private Map<String,Goods> map;        

    public Map<String, Goods> getMap() {
        return map;
    }

    public void setMap(Map<String, Goods> map) {
        this.map = map;
    }

    public String execute(){
        for (String key : map.keySet()) {
            Goods goods = map.get(key);        
            System.out.println(key + "\t" + goods);
        }
        return NONE;
    }
}
