源码

TecentZhaoPin.py

    # -*- coding: utf-8 -*-
import scrapy
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from TecentSpider.items import TecentspiderItem

class TecentzhaopinSpider(CrawlSpider):
    name = 'TecentZhaoPin'
    allowed_domains = ['hr.tencent.com']
    start_urls = ['https://hr.tencent.com/position.php?keywords=python']

    rules = (
        Rule(LinkExtractor(allow=r'.+&start=\d'), follow=True),       # 下一页的获取
        Rule(LinkExtractor(allow=r'.+\d&keywords=python&tid=0&lid=0'),callback='parse_item',follow=False), # 详情页面的URL获取
    )

    def parse_item(self, response):
        job_title = response.xpath("//*[@id='sharetitle']/text()").get()     # 职位名
        require = response.xpath("/html/body/div[3]/div/table/tr[4]/td/ul//text()").getall()      # 职位要求\
        site = response.xpath("/html/body/div[3]/div/table/tr[@class='c bottomline']/td[1]/text()").get()   # 工作地点
        people_num = response.xpath('/html/body/div[3]/div/table/tr[2]/td[3]/text()').get()     # 招聘人数
        print(f"职位名:{job_title}\n职位要求{require}\n工作地点{site}\n招聘人数{people_num}")
        item = TecentspiderItem(job_title=job_title,require=require,site=site,people_num=people_num)
        yield item

pipelines.py

from scrapy.exporters import JsonLinesItemExporter


class TecentspiderPipeline(object):

    def __init__(self):
        self.fileProject = open("腾讯Python职位招聘信息.json", 'wb')  # json文件对象
        self.exporters = JsonLinesItemExporter(self.fileProject, ensure_ascii=False, encoding='utf-8')

    def process_item(self, item, spider):
        self.exporters.export_item(item)
        return item

    def close_spider(self, spider):
        self.fileProject.close()
