resold_Apartment.py

# -*- coding: utf-8 -*-
import scrapy
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from XuChangHouse.items import XuchanghouseItem

class ResoldApartmentSpider(CrawlSpider):
    name = 'resold_Apartment'
    allowed_domains = ['xuchang.anjuke.com']
    start_urls = ['https://xuchang.anjuke.com/sale/p1-rd1/#filtersort']

    rules = (
        Rule(LinkExtractor(allow=r'https://xuchang.anjuke.com/sale/.+/#filtersort'), follow=True), # 匹配页数
        Rule(LinkExtractor(allow=r'.+[A-Z0-9]?from=.*',),callback="parse_item",follow=False),     # 详情页面
    )

    def parse_item(self, response):
        estate = response.xpath("//div[@class='houseInfo-content']/a/text()").get()  # 所属小区
        total = response.css('.light > em:nth-child(1)::text').get()  # 总价
        down_payment = response.xpath("//ul/li[6]/div[2]/text()").get().strip()  # 首付
        unit_price = response.xpath("//ul/li[3]/div[2]/text()").get().strip()  # 房屋单价
        area = response.xpath("//ul/li[5]/div[2]/text()").get().strip()  # 建筑面积
        family_type = response.xpath("//ul/li[2]/div[2]/text()").get().replace('\t', '')  # 户型
        time = response.xpath("//ul/li[7]/div[2]/text()").get().strip() # 建造年代
        house_types = response.xpath('//ul/li[10]/div[2]/text()').get().strip()       # 房屋类型
        age_limit = response.xpath("//ul/li[13]/div[2]/text()").get().strip()       # 房本年限
        # Monthly_payment = response.css("#reference_monthpay").get() # 月供,span标签在页面上显示但是scrapy提取之后没有内容

        # print(f"小区{estate}\n年代{time}\n类型{house_types}\n房本年限{age_limit}\n户型{family_type}面积{area}单价{unit_price}\n首付{down_payment}\n")
        item = XuchanghouseItem(estate=estate,total=total,down_payment=down_payment,unit_price=unit_price,area=area,family_type=family_type,time=time,house_types=house_types,age_limit=age_limit)
        yield item

pipelines.py

# -*- coding: utf-8 -*-

# Define your item pipelines here
#
# Don't forget to add your pipeline to the ITEM_PIPELINES setting
# See: https://doc.scrapy.org/en/latest/topics/item-pipeline.html

from scrapy.exporters import JsonLinesItemExporter


class XuchanghousePipeline(object):

    def __init__(self):
        self.fileProject = open("XuChangHouse.json", 'wb')  # json文件对象
        self.exporters = JsonLinesItemExporter(self.fileProject, ensure_ascii=False, encoding='utf-8')

    def process_item(self, item, spider):
        self.exporters.export_item(item)
        return item

def close_spider(self, spider):
    self.fileProject.close()
