网站分析

网站首页中有各种药品分类,打开分类页面我们可以看到各种的药品分类信息,分析它的URL规则,

https://www.111.com.cn/categories/953785-j1
https://www.111.com.cn/categories/953787-j1
https://www.111.com.cn/categories/956104-j1

编写正则
categories/\d{6}-j\d

爬取规则

# -*- coding: utf-8 -*-
import scrapy
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from scrapy_splash import SplashRequest
from YaoPinSpider.items import YaopinspiderItem


class YaopinSpider(CrawlSpider):
    name = 'YaoPin'  # 爬虫名字
    allowed_domains = ['111.com.cn']  # 爬取的域
    start_urls = ['https://www.111.com.cn/categories/']  # 开始爬取的URL

    # 抓取规则
    rules = (
        Rule(LinkExtractor(allow=r'categories/\d{6}-j\d'), callback="parse_next", follow=True),  # 匹配药品分类
        Rule(LinkExtractor(allow=r'product/\d+.html'), callback="parse_item", follow=True)  # 匹配药品详情页面
    )

    splash_args = {
        "wait": 0.5,  # Splash执行完成之后会等待一段时间后返回
    }

    def parse_next(self, response):
        # 下一页的处理
        next_url = response.xpath(
            "//div[@id='search_table']/div[@class='turnPageBottom']/a[contains(@class,'page_next')]/@href").extract_first()

        if next_url:
            next_url = "https://" + next_url[2:]
            yield SplashRequest(url=next_url, callback=self.parse_item, args=self.splash_args, endpoint="render.html")
        else:
            print(f"{response.url}没有下一页")

    def parse_item(self, response):
        """
        将CrawlSpider匹配的url交给Splash处理
        :param response:
        :return:
        """
        url = response.url
        yield SplashRequest(url=url, callback=self.parse_detail, args=self.splash_args, endpoint="render.html")

    def parse_detail(self, response):
        """
        解析Splash渲染完成之后的页面
        :param response:
        :return:
        """
        items = YaopinspiderItem()  # 实例化item类
        tr_list = response.xpath("//div[@class='goods_intro']/table/tbody")
        for tr in tr_list:
            drug_name = tr.xpath("./tr[1]/td/text()").extract_first()  # 药品名
            brand = tr.xpath("./tr[2]/td[1]/text()").extract_first()  # 品牌
            spec = tr.xpath("./tr[2]/td[2]/text()").extract_first()  # 规格
            weight = tr.xpath("./tr[3]/td[1]/text()").extract_first()  # 重量
            manufacturer = tr.xpath("./tr[3]/td[2]/text()").extract_first()  # 生产厂商
            approval_num = tr.xpath("./tr[4]/td/text()").extract_first().strip().strip("(")  # 批准文号
            items['drug_name'] = drug_name
            items['brand'] = brand
            items['spec'] = spec
            items['weight'] = weight
            items['manufacturer'] = manufacturer
            items['approval_num'] = approval_num
            # print_format = f"药品名字:{drug_name},品牌:{brand},规格:{spec},重量:{weight},生产厂商:{manufacturer},批准文号:{approval_num}"
            # print(print_format)

        price = response.xpath(
            "//div[@class='shangpin_info']/dl[@class='clearfix']/dd/span[@class='good_price']/text()").extract_first()  # 药品价格
        # print(price)
        items['price'] = price
        items["detail_url"] = response.url
        yield items

class MysqlPipeline(object):
    '''
    将数据保存到Mysql
    这是同步的插入方式
    这种方式当数据量过大的时候会引起阻塞
    '''

    def __init__(self):
        self.conn = MySQLdb.connect('localhost', 'root', 'LiuChen0.', 'yaopindata', charset='utf8',
                                    use_unicode=True)  # 数据库连接
        self.cursor = self.conn.cursor()  # 获取游标

    def process_item(self, item, spider):
        insert_sql = """
                    insert into drug(drug_name,brand,spec,weight,manufacturer,approval_num,price,detail_url)
                    values(%s,%s,%s,%s,%s,%s,%s,%s)
                    """
        self.cursor.execute(insert_sql,
                    (item['drug_name'], item['brand'], item['spec'], item['weight'], item['manufacturer'],
                        item['approval_num'], item['price'], item['detail_url']))
        self.conn.commit()
