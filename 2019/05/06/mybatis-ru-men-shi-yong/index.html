在Maven中配置Mybatis和其他坐标

我这里配置的有Junit,Mybatis,SQL(注意版本号哦)

<?xml version="1.0" encoding="UTF-8"?>
<project xmlns="http://maven.apache.org/POM/4.0.0"
        xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
        xsi:schemaLocation="http://maven.apache.org/POM/4.0.0 http://maven.apache.org/xsd/maven-4.0.0.xsd">
    <modelVersion>4.0.0</modelVersion>

    <groupId>com.chen</groupId>
    <artifactId>day1_mybaties1</artifactId>
    <version>1.0-SNAPSHOT</version>
    <!--打包方式-->
    <packaging>jar</packaging>

    <!--导入坐标-->
    <dependencies>
        <!--MyBitais坐标-->
        <dependency>
            <groupId>org.mybatis</groupId>
            <artifactId>mybatis</artifactId>
            <version>3.4.5</version>
        </dependency>
        <!--Mysql数据库坐标,根据自己的Mysql数据库版本填写-->
        <dependency>
            <groupId>mysql</groupId>
            <artifactId>mysql-connector-java</artifactId>
            <version>8.0.15</version>
        </dependency>
        <!--log4j-->
        <dependency>
            <groupId>log4j</groupId>
            <artifactId>log4j</artifactId>
            <version>1.2.12</version>
        </dependency>
        <!--单元测试junit-->
        <dependency>

            <version>4.10</version>
        </dependency>
    </dependencies>
</project>

package domain;

import java.io.Serializable;

    /**
    * @program: day1_mybaties1
    * @description: 用户的实体类

创建Dao接口

定义一个查询所有的方法

package dao;

import domain.User;
import org.apache.ibatis.annotations.Select;

import java.util.List;

public interface UserDao {
    /**
    * @Description: 查询所有的方法
    */
    List<User> findAll();
}

创建Mybatis的主配置文件(SqlMapConfig.xml)

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE configuration
        PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-config.dtd">
<!--Mybatis主配置文件-->
<configuration>
    <!--配置环境-->
    <environments default="mysql">
        <!--配置mysql环境-->
        <environment id="mysql">
            <!--配置事务的类型-->
            <transactionManager type="JDBC"></transactionManager>
            <!--配置连接池pooled,连接池有很多,这里使用的是pooled-->
            <dataSource type="POOLED">
                <!--配置连接数据库的基本信息-->
                <property name="driver" value="com.mysql.cj.jdbc.Driver"/>
                <property name="url"
                        value="jdbc:mysql://localhost:3306/chuanzhijava?useUnicode=true&amp;characterEncoding=utf-8&amp;useSSL=false&amp;serverTimezone=UTC"/>
                <property name="username" value="root"/>
                <property name="password" value="LiuChen0."/>
            </dataSource>
        </environment>
    </environments>

    <!--指定映射配置文件的位置,映射配置文件指的是每个DAO独立的配置文件-->
    <mappers>
        <mapper resource="dao/UserDao.xml"/>
    </mappers>
</configuration>

创建映射配置文件

创建位置：必须和持久层接口在相同的包中。
名称：必须以持久层接口名称命名文件名，扩展名是.xml

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="dao.UserDao">
    <!--配置查询所有-->
    <select id="findAll" resultType="domain.User">
        select * from user
    </select>
</mapper>

注意事项

在idea中创建目录的时候，目录和包是不一样的

包在创建时：com.chen.dao它是三级结构
目录在创建时：com.chen.dao是一级目录

mybatis的映射配置文件位置必须和dao接口的包结构相同

映射配置文件的mapper标签namespace属性的取值必须是dao接口的全限定类名

<mapper namespace="dao.UserDao">
    <!--配置查询所有-->
    <select id="findAll" resultType="domain.User">
        select * from user
    </select>
</mapper>

映射配置文件的操作配置（select），id属性的取值必须是dao接口的方法名

<select id="findAll" resultType="domain.User">
        select * from user
</select>

编写测试类

package test;

import domain.User;
import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import dao.UserDao;

import java.io.InputStream;
import java.util.List;

public class MyBatisTest {
    /**
    * @program: day1_mybaties1
    * @description: 第一个Mybatis测试类
    * @author: Mr.Wang
    * @create: 2019-05-06 14:03
    **/
    //入门案例
    public static void main(String[] args) throws Exception {
        //1.读取配置文件
        InputStream in = Resources.getResourceAsStream("SqlMapConfig.xml");
        //2.创建sqlSessionFactory
        SqlSessionFactoryBuilder builder = new SqlSessionFactoryBuilder();
        //3.使用构建者创建工厂对象 SqlSessionFactory
        SqlSessionFactory factory = builder.build(in);
        //4.使用 SqlSessionFactory 生产SqlSession 对象
        SqlSession session = factory.openSession();
        //5.使用 SqlSession 创建 dao 接口的代理对象
        UserDao userdao = session.getMapper(UserDao.class);
        //6.使用代理对象执行查询所有方法
        List<User> users = userdao.findAll();
        for (User user : users
        ) {
            System.out.println(user.getId());
            System.out.println(user.getUsername());
            System.out.println(user.getAddress());
            System.out.println(user.getSex());
            System.out.println("--------------------");
        }
        //7.释放资源
        session.close();
        in.close();
    }
}

基于注解的MyBatis使用

在持久层接口中添加注解

在dao接口的方法上使用@Select注解，并且指定SQL语句

public interface UserDao {
    /**
    * @Description: 查询所有的接口
    */
    @Select("select * from user")
    List<User> findAll();
}

修改SqlMapConfig.xml

在SqlMapConfig.xml中的mapper配置时，使用class属性指定dao接口的全限定类名

注解方式
<mappers>
        <mapper class="dao.UserDao"/>
</mappers>

注意在使用基于注解的 Mybatis 配置时，务必要移除 xml的映射配置（UserDao.xml）。
